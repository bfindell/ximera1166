\begin{problem}
Write an expression for the area of the triangle below (not drawn to scale).   
\begin{image}
\begin{tikzpicture}
\draw [line width=0.8pt,dash pattern=on 2pt off 2pt] (0.,0.)-- (0.,2.5)-- (5.,2.5)-- (5.,0.)--(2.,0.);
\draw [line width=0.8pt] (0.,0.)-- (5.,2.5)-- (2.,0.)-- cycle;
\draw (1.1,-0.2) node {$a$};
\draw (3.5,-0.2) node {$x$};
\draw (5.2,1.3) node {$5$};
\draw [color=white] (-4.,0.) circle (0.2pt);
\draw [color=white] (8.,0.) circle (0.2pt);
\end{tikzpicture}
\end{image}
Area = $\answer{5a/2}$. 
\begin{problem}
Supposing the base and the height are fixed, what can we conclude from the fact that the answer doesn't include $x$? 
\begin{selectAll}
\choice{The answer should include an $x$.}
\choice[correct]{The area is independent of $x$.}
\choice{The area increases if $x$ increases.}
\choice{The area decreases if $x$ decreases.}
\choice[correct]{The extended portion of the base can be as long or as short as you want and the area remains the same.}
\end{selectAll}
\end{problem}
\end{problem}
