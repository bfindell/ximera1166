\begin{problem}
Recall that a trapezoid is a quadrilateral with two parallel sides. Consider the following picture:
\begin{image}
%\includegraphics{trap.png}
\begin{tikzpicture}[scale=0.8]
\draw [line width=0.8pt] (0.,0.)-- (2.,3.) -- (6.,3.)-- (7.,0.) -- cycle;
\draw [line width=0.8pt] (0.,0.)-- (6.,3.);
\draw [fill=white] (-4.,0.) circle (0.2pt);
\draw [fill=white] (11.,0.) circle (0.2pt);
\end{tikzpicture}
\end{image}
How does the above picture prove that the area of a trapezoid is
\[
\mathrm{area} = \frac{h(b_1 + b_2)}{2}
\]
where $h$ is the height of the trapezoid and $b_1$ and $b_2$ are the lengths of the parallel sides?

\begin{freeResponse}
\end{freeResponse}

I \wordChoice{\choice[correct]{have}\choice{have not}} entered my own answer in the box above.  

\begin{problem}
Explanation:  Using the parallel sides as bases, the area of the trapezoid is the $\answer[format=string]{sum}$ of the areas of two triangles that have different bases but the same $\answer[format=string]{height}$ as the trapezoid.  

\begin{image}
\begin{tikzpicture}[scale=0.8]
\draw [line width=0.8pt] (0.,0.)-- (2.,3.) -- (6.,3.)-- (7.,0.) -- cycle;
\draw [line width=0.8pt] (0.,0.)-- (6.,3.);
\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (6.,0.) -- (6.,3.);
\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (0.,0.) -- (0.,3.) -- (2.,3.);
\draw (3.5,-0.3) node {$b_1$};
\draw (3.7,3.3) node {$b_2$};
\draw (5.7,1.4) node {$h$};
\draw (-0.3,1.4) node {$h$};
\draw [fill=white] (-4.,0.) circle (0.2pt);
\draw [fill=white] (11.,0.) circle (0.2pt);
\end{tikzpicture}
\end{image}


Note:  Use an underscore character for a subscript: b\textunderscore 1.   
\[
\mathrm{Area} =  \frac{hb_1}{2} + \frac{hb_2}{2}= \frac{h(b_1 + b_2)}{2}
\]
\end{problem}
\end{problem}

\begin{problem}
Explain how the following picture ``proves'' that the area of a
  parallelogram is base times height.
\begin{image}
%\includegraphics{para3.png}
\begin{tikzpicture}[scale=0.8]
\draw [line width=0.8pt] (0.,0.) -- (2.,3.) -- (9.,3.) -- (7.,0.) -- cycle;
\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (7.,0.)-- (9.,0.)-- (9.,3.);
\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (2.,3.)-- (2.,0.);
\draw [fill=white] (-4.,0.) circle (0.2pt);
\draw [fill=white] (13.,0.) circle (0.2pt);
\end{tikzpicture}
\end{image}
Imagine cutting off the $\answer[format=text]{triangle}$ on the left and placing it in the (congruent) dotted region on the right.  Then we have a $\answer[format=string]{rectangle}$ with the same base and height as the parallelogram.  The cutting and rearranging doesn't change the $\answer[format=string]{area}$ of the figure, so the area of the parallelogram is the same as the area of the rectangle, which is base times height. 
\end{problem}

\begin{problem}
This parallelogram problem is much like Giorgio's triangle problem: (1) compute the area of a larger figure (a rectangle) and then (2) subtract the areas we don't want (two congruent right triangles with base $x$): 
\begin{image}
\begin{tikzpicture}[scale=0.8]
%\includegraphics{paragiorgio.png}
\draw [line width=0.8pt] (0.,0.)-- (5.,3.) -- (8.,3.)-- (3.,0.) -- cycle;
\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (3.,0.) -- (8.,0.) -- (8.,3.);
%\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (5.,3.)-- (5.,0.);
\draw [line width=0.8pt,dash pattern=on 3pt off 3pt] (0.,0.) -- (0.,3.) -- (5.,3.);
\draw (1.5,-0.2) node {$b$};
\draw (5.5,-0.2) node {$x$};
\draw (8.25,1.4) node {$h$};
\draw [fill=white] (-4.,0.) circle (0.2pt);
\draw [fill=white] (12.,0.) circle (0.2pt);
\end{tikzpicture}
\end{image}
 
\[
\mathrm{area} =  \answer{h(b+x)} - \answer{\frac{hx}{2}} - \answer{\frac{hx}{2}}= hb + hx - \frac{hx}{2}- \frac{hx}{2} = \answer{bh}.
\]
